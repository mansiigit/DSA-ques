#include<iostream>
using namespace std;
int bs(int a[],int start,int end ,int x)
{
    int j;
   while(start<=end)
    {
        int mid=start+(end-start)/2;
        if(a[mid]==x)
        {
            return mid;
        }
        else
        {
            if(a[mid]>x)
            {
                end=mid-1;
            }
            else
            {
                start=mid+1;
            }
        }
    }
    return -1;
}
int expo_search(int a[],int n,int x)
{

    if(a[0]==x)
    {
        return 0;
    }
    int i=1;
    while(i<=n &&  a[i]<=x)
    {
        i=i*2;
    }
    return bs(a,i/2,min(i,n-1),x);
}
int main()
{
    int a[]={1,2,3,4,5,6,7,8,9};
    int n=sizeof(a)/ sizeof(int);
    int x=8;
    int ans=expo_search(a,n,x);
    cout << ans << endl;
    return 0;
}
